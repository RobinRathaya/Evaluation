CREATE TABLE EV_EMPLOYEE(
ID INT ,
NAME VARCHAR(50),
EMAIL VARCHAR(50),
PASSWORD VARCHAR(50),
PRIMARY KEY(ID)
);

CREATE TABLE EV_EMPLOYEETOPICS(
ID INT AUTO_INCREMENT,
EMPID INT ,
TOPICID INT ,
STATUSID INT,
CREATEDON DATETIME,
MODIFIEDON DATETIME,
PRIMARY KEY(ID),
FOREIGN KEY(EMPID) REFERENCES EV_EMPLOYEE(ID),
FOREIGN KEY(TOPICID) REFERENCES EV_TOPICS(TOPICID),
FOREIGN KEY(STATUSID) REFERENCES EV_STATUS(ID)
);
